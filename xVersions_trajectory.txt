#!/usr/bin/env python3
import rospy
import math
from geometry_msgs.msg import Twist
from nav_msgs.msg import Odometry
from tf.transformations import euler_from_quaternion

class MoveRobot:
    def __init__(self):
        rospy.init_node("move_robot_with_odom")

        # Publisher for velocity commands
        self.pub = rospy.Publisher("/cmd_vel", Twist, queue_size=10)
        # Subscriber for odometry data
        rospy.Subscriber("/odom", Odometry, self.odom_callback)

        self.rate = rospy.Rate(20)  # 20 Hz control loop

        # Odometry state variables
        self.x = 0.0
        self.y = 0.0
        self.yaw = 0.0

    def odom_callback(self, msg):
        """Read odometry and extract position and yaw angle."""
        self.x = msg.pose.pose.position.x
        self.y = msg.pose.pose.position.y

        # Convert quaternion to yaw (Euler)
        q = msg.pose.pose.orientation
        (_, _, self.yaw) = euler_from_quaternion([q.x, q.y, q.z, q.w])

    # ------------------------
    # Basic motion primitives
    # ------------------------
    def move_distance(self, distance, speed=0.2):
        """Move straight a specific distance."""
        start_x, start_y = self.x, self.y
        cmd = Twist()

        while not rospy.is_shutdown():
            dx = self.x - start_x
            dy = self.y - start_y
            dist = math.sqrt(dx**2 + dy**2)

            if dist >= abs(distance):
                break

            cmd.linear.x = speed if distance > 0 else -speed
            cmd.angular.z = 0.0
            self.pub.publish(cmd)
            self.rate.sleep()

        # Stop
        cmd.linear.x = 0.0
        self.pub.publish(cmd)
        rospy.loginfo(f"Moved {dist:.2f} m")

    def rotate_angle(self, angle_deg, angular_speed_deg=30):
        """Rotate a specific angle in degrees."""
        angle_rad = math.radians(angle_deg)
        angular_speed = math.radians(angular_speed_deg)

        start_yaw = self.yaw
        cmd = Twist()
        direction = 1 if angle_rad > 0 else -1

        while not rospy.is_shutdown():
            delta_yaw = self.yaw - start_yaw
            delta_yaw = math.atan2(math.sin(delta_yaw), math.cos(delta_yaw))

            if abs(delta_yaw) >= abs(angle_rad):
                break

            cmd.linear.x = 0.0
            cmd.angular.z = direction * angular_speed
            self.pub.publish(cmd)
            self.rate.sleep()

        cmd.angular.z = 0.0
        self.pub.publish(cmd)
        rospy.loginfo(f"Rotated {angle_deg}°")

    def move_arc(self, radius, angle_deg, speed=0.2):
        """Drive along a circular arc with given radius and angle."""
        angle_rad = math.radians(angle_deg)
        angular_speed = speed / radius  # ω = v / R
        direction = 1 if angle_deg > 0 else -1

        start_yaw = self.yaw
        cmd = Twist()

        while not rospy.is_shutdown():
            delta_yaw = self.yaw - start_yaw
            delta_yaw = math.atan2(math.sin(delta_yaw), math.cos(delta_yaw))

            if abs(delta_yaw) >= abs(angle_rad):
                break

            cmd.linear.x = speed
            cmd.angular.z = direction * angular_speed
            self.pub.publish(cmd)
            self.rate.sleep()

        cmd.linear.x = 0.0
        cmd.angular.z = 0.0
        self.pub.publish(cmd)
        rospy.loginfo(f"Moved in arc (R={radius:.2f} m, angle={angle_deg}°)")

    # ------------------------
    # Feedback control: go-to-goal
    # ------------------------
    def go_to_goal(self, xg, yg, Kv=0.4, Kw=2.0, threshold=0.1):
        """Improved go-to-goal controller with orientation alignment."""
        cmd = Twist()
        rospy.loginfo(f"Starting go_to_goal to ({xg:.2f}, {yg:.2f})")

        while not rospy.is_shutdown():
            # Compute position error
            ex = xg - self.x
            ey = yg - self.y
            ed = math.sqrt(ex**2 + ey**2)

            # Desired heading to goal
            e_theta = math.atan2(ey, ex) - self.yaw
            e_theta = math.atan2(math.sin(e_theta), math.cos(e_theta))  # wrap [-pi, pi]

            # Stop condition
            if ed < threshold:
                break

            # --- Step 1: Align orientation first ---
            if abs(e_theta) > 0.3:  # about 17 degrees
                V = 0.0              # stop forward motion
                W = 1.5 * e_theta    # strong rotation to face goal
            else:
                # --- Step 2: Move toward goal while correcting heading ---
                V = Kv * ed
                W = Kw * e_theta

            # --- Step 3: Limit velocities for stability ---
            V = max(min(V, 0.25), -0.25)
            W = max(min(W, 1.5), -1.5)

            cmd.linear.x = V
            cmd.angular.z = W
            self.pub.publish(cmd)
            self.rate.sleep()

        # Stop when goal reached
        cmd.linear.x = 0.0
        cmd.angular.z = 0.0
        self.pub.publish(cmd)
        rospy.loginfo(f"Arrived near goal ({xg:.2f}, {yg:.2f})")

if __name__ == "__main__":
    robot = MoveRobot()
    rospy.sleep(1)  # Wait for odometry to start

    try:
        rospy.loginfo("=== Starting trajectory ===")
        
        robot.go_to_goal(0.0, 0.0)   # Home to (0,0)
        rospy.sleep(0.5)

        robot.go_to_goal(0.0, 2.0)   # Move to (0,2)
        rospy.sleep(0.5)

        robot.go_to_goal(2.0, 2.0)   # Move to (2,2)
        rospy.sleep(0.5)
               
        robot.go_to_goal(2.0, 0.0, Kv=0.7, Kw=2.0)   # Move to (2,0)
        rospy.sleep(0.5)

        #robot.go_to_goal(2.0, 1.0, Kv=0.7, Kw=2.0)

        robot.go_to_goal(0.0, 0.0, Kv=0.7, Kw=2.0)   # Then to (start)
        rospy.sleep(0.5)
        
        robot.go_to_goal(0.0, 0.0)   # Finally return home

        rospy.loginfo("Trajectory complete ✅")

    except rospy.ROSInterruptException:
        pass
